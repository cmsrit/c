import java.util.Arrays;

public class Lab1 {
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);
        // 1a. Quadratic Roots
        System.out.println("Enter a, b, c for ax^2 + bx + c:");
        double a = sc.nextDouble(), b = sc.nextDouble(), c = sc.nextDouble();
        double d = b * b - 4 * a * c;
        if (d > 0) System.out.println("Roots: " + (-b + Math.sqrt(d))/(2*a) + ", " + (-b - Math.sqrt(d))/(2*a));
        else if (d == 0) System.out.println("Root: " + (-b / (2 * a)));
        else System.out.println("Roots are imaginary.");

        // 1b. Matrix Multiplication (2x2 example for brevity)
        int[][] m1 = {{1, 2}, {3, 4}}, m2 = {{1, 0}, {0, 1}}, res = new int[2][2];
        System.out.println("Matrix Multiplication Result:");
        for (int i = 0; i < 2; i++) {
            for (int j = 0; j < 2; j++) {
                for (int k = 0; k < 2; k++) res[i][j] += m1[i][k] * m2[k][j];
                System.out.print(res[i][j] + " ");
            }
            System.out.println();
        }

        // 1c. Bubble Sort
        int[] arr = {5, 1, 4, 2, 8};
        for (int i = 0; i < arr.length - 1; i++)
            for (int j = 0; j < arr.length - i - 1; j++)
                if (arr[j] > arr[j + 1]) { // Change > to < for descending
                    int temp = arr[j]; arr[j] = arr[j+1]; arr[j+1] = temp;
                }
        System.out.println("Sorted: " + Arrays.toString(arr));
    }
}

3. Complex NumbersRequirement: Encapsulation, constructors, Add, Sub, Compare5.Explanation: A class Complex holds real and imaginary parts. Methods return new Complex objects for arithmetic operations.Javaclass Complex {
    private double real, img;
    public Complex(double r, double i) { real = r; img = i; }
    
    public Complex add(Complex c) { return new Complex(real + c.real, img + c.img); }
    public Complex sub(Complex c) { return new Complex(real - c.real, img - c.img); }
    public void display() { System.out.println(real + " + i" + img); }
    
    public boolean isEqual(Complex c) { return real == c.real && img == c.img; }

    public static void main(String[] args) {
        Complex c1 = new Complex(4, 5);
        Complex c2 = new Complex(2, 3);
        c1.add(c2).display(); // 6.0 + i8.0
        c1.sub(c2).display(); // 2.0 + i2.0
    }
}

5. Polymorphism (String Compare)Requirement: Overload usrstrcmp: one compares full strings, one compares n chars7.Explanation: Method overloading allows two methods with the same name but different parameters.Javapublic class Lab5 {
    // Version 1: Compare full strings
    static int usrstrcmp(String s1, String s2) {
        return s1.compareTo(s2);
    }
    
    // Version 2: Compare first n characters
    static int usrstrcmp(String s1, String s2, int n) {
        return s1.substring(0, n).compareTo(s2.substring(0, n));
    }

    public static void main(String[] args) {
        String s1 = "Apple", s2 = "Application";
        System.out.println("Full compare: " + usrstrcmp(s1, s2)); // Non-zero (Not equal)
        System.out.println("First 3 chars: " + usrstrcmp(s1, s2, 3)); // 0 (Equal "App")
    }
}

Part B7. Producer-ConsumerRequirement: Multi-threaded program illustrating producer-consumer9.Explanation: Uses wait() and notify() to synchronize threads. The producer waits if the buffer is full, and the consumer waits if it is empty.Javaclass Buffer {
    int data;
    boolean available = false;
    
    synchronized void produce(int val) {
        if (available) try { wait(); } catch (Exception e) {}
        data = val; available = true;
        System.out.println("Produced: " + val);
        notify();
    }
    
    synchronized void consume() {
        if (!available) try { wait(); } catch (Exception e) {}
        System.out.println("Consumed: " + data);
        available = false;
        notify();
    }
}

public class Lab7 {
    public static void main(String[] args) {
        Buffer b = new Buffer();
        new Thread(() -> { for(int i=0; i<5; i++) b.produce(i); }).start();
        new Thread(() -> { for(int i=0; i<5; i++) b.consume(); }).start();
    }
}

import java.sql.*;
import java.util.Scanner;

public class JDBCCrud {
    // Update these credentials as per your local setup
    static final String URL = "jdbc:mysql://localhost:3306/demo";
    static final String USER = "root"; 
    static final String PASS = "password"; 

    public static void main(String[] args) {
        try (Connection conn = DriverManager.getConnection(URL, USER, PASS);
             Scanner sc = new Scanner(System.in)) {
            while (true) {
                System.out.println("1.Insert 2.Select 3.Update 4.Delete 5.Exit");
                int ch = sc.nextInt();
                if (ch == 5) break;
                
                Statement stmt = conn.createStatement();
                switch (ch) {
                    case 1: // Insert
                        System.out.println("Enter ID, Name, Age:");
                        String sql1 = "INSERT INTO students VALUES ("+sc.nextInt()+", '"+sc.next()+"', "+sc.nextInt()+")";
                        stmt.executeUpdate(sql1);
                        break;
                    case 2: // Select
                        ResultSet rs = stmt.executeQuery("SELECT * FROM students");
                        while (rs.next()) System.out.println(rs.getInt(1) + " " + rs.getString(2) + " " + rs.getInt(3));
                        break;
                    case 3: // Update
                        System.out.println("Enter ID to update and new Name:");
                        stmt.executeUpdate("UPDATE students SET name='" + sc.next() + "' WHERE id=" + sc.nextInt());
                        break;
                    case 4: // Delete
                        System.out.println("Enter ID to delete:");
                        stmt.executeUpdate("DELETE FROM students WHERE id=" + sc.nextInt());
                        break;
                }
            }
        } catch (Exception e) { e.printStackTrace(); }
    }
}

import java.io.*;
import javax.servlet.*;
import javax.servlet.http.*;

public class VoterSrv extends HttpServlet {
    public void doGet(HttpServletRequest req, HttpServletResponse res) throws IOException {
        res.setContentType("text/html");
        PrintWriter out = res.getWriter();
        
        String name = req.getParameter("name");
        int age = Integer.parseInt(req.getParameter("age"));
        
        if (age < 18) 
            out.println("<h2 style='color:red'>Hello " + name + ", you are not authorized to visit the site.</h2>");
        else 
            out.println("<h2 style='color:green'>Welcome " + name + " to this site.</h2>");
            
        out.println("<a href='index.html'>Back</a>");
    }
}
