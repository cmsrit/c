package AdvMath;

public class MathsEquation {
    double x;
    int n;

    public MathsEquation(double x) {
        this.x = x;
    }

    public MathsEquation(int n) {
        this.n = n;
    }

    public void Trigo() {
        System.out.println("Y=" + (Math.sin(x) + Math.cos(x) + Math.tan(x)));
    }
}
