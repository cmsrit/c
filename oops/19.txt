import javax.swing.*;

public class SwingEx1 extends JFrame {

    SwingEx1() {
        JButton b = new JButton("Click");
        b.setBounds(130, 100, 100, 40);

        add(b);
        setSize(400, 500);
        setLayout(null);
        setVisible(true);
        setDefaultCloseOperation(JFrame.EXIT_ON_CLOSE);
    }

    public static void main(String[] args) {
        new SwingEx1();
    }
}
